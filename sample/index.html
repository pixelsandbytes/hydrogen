Inheritance

function Watch(color) {
    this.color = color;
}
h.create(Watch, {
    getLabel: function() {
        return 'Watch';
    },
    getLongLabel: function() {
        return this.getLabel() + ' (' + this.color + ')';
    }
});

function AnalogWatch(color, band) {
    this.super.constructor.call(this, color);
    this.band = band;
}
h.create(AnalogWatch, Watch, {
    getLabel: function() {
        return 'Analog Watch';
    }
});

Private Variables

function Counter() {}
h.attach(Counter, function() {
    var count = 0;
    return {
        get: function() {
            return count;
        },
        increment: function() {
            count++;
        }
    }
});

Private Variables & Functions

function CounterByMultiple(multiple) {
    this.multiple = multiple;
}
h.create(CounterByMultiple, Counter, {})
.attach(CounterByMultiple, function() {
    var getCount = this.get; // preserve Counter.get()

    function getMultipliedCount() {
        return this.multiple * getCount();
    }

    return {
        get: function() {
            return 'Count: ' + getMultipliedCount.call(this);
        }
    }
});
